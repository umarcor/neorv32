168#define CUSTOM_INSTR_R2_TYPE(funct7, funct5, rs1, funct3, opcode) \
169({ \
170 uint32_t __return; \
171 asm volatile ( \
172 "" \
173 : [output] "=r" (__return) \
174 : [input_i] "r" (rs1) \
175 ); \
176 asm volatile( \
177 ".word ( \
178 (((" #funct7 ") & 0x7f) << 25) | \
179 (((" #funct5 ") & 0x1f) << 20) | \
180 ((( regnum_%1 ) & 0x1f) << 15) | \
181 (((" #funct3 ") & 0x07) << 12) | \
182 ((( regnum_%0 ) & 0x1f) << 7) | \
183 (((" #opcode ") & 0x7f) << 0) \
184 );" \
185 : [rd] "=r" (__return) \
186 : "r" (rs1) \
187 ); \
188 __return; \
189})
